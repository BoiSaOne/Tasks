1) SELECT TOP (1) E.Id EmpleyeeId, E.Name EmployeeName, E.Salary FROM Employee E 
   ORDER BY E.Salary DESC
2) WITH Recursives AS (SELECT E.Id, E.ChiefId, 0 CountStep FROM Employee E 
   WHERE ChiefId IS NULL
   UNION ALL
   SELECT T.Id, T.ChiefId, R.CountStep+1 FROM Employee T 
   INNER JOIN Recursives R ON T.ChiefId = R.Id)
   SELECT Max(CountStep) MaxStep FROM Recursives
        
3) SELECT TOP (1) D.Id DepartmentId, D.Name DepartmentName, SUM(E.Salary) Sum FROM Department D 
   LEFT JOIN Employee E ON E.DepartmentId = D.Id 
   GROUP BY D.Id, D.Name 
   ORDER BY SUM(E.Salary) DESC
